Условные операторы

Использование условных операторов if-else:

Примеры условий

int sensorValue = analogRead(A0);

// Простое условие
if (sensorValue > 500) {
    digitalWrite(13, HIGH);
} else {
    digitalWrite(13, LOW);
}

// Множественные условия
if (sensorValue < 200) {
    // Низкий уровень
    digitalWrite(2, HIGH);
} else if (sensorValue < 800) {
    // Средний уровень
    digitalWrite(3, HIGH);
} else {
    // Высокий уровень
    digitalWrite(4, HIGH);
}

Циклы

Различные типы циклов:

Примеры циклов

// Цикл for
for (int i = 0; i < 10; i++) {
    digitalWrite(ledPin, HIGH);
    delay(100);
    digitalWrite(ledPin, LOW);
    delay(100);
}

// Цикл while
int count = 0;
while (count < 5) {
    digitalWrite(ledPin, HIGH);
    delay(1000);
    digitalWrite(ledPin, LOW);
    delay(1000);
    count++;
}

// Цикл do-while
do {
    sensorValue = analogRead(A0);
    delay(100);
} while (sensorValue > 100);

Функции

Создание и использование пользовательских функций:

Примеры функций

// Функция без параметров и возвращаемого значения
void blinkLED() {
    digitalWrite(13, HIGH);
    delay(1000);
    digitalWrite(13, LOW);
    delay(1000);
}

// Функция с параметрами
void blinkTimes(int pin, int times) {
    for (int i = 0; i < times; i++) {
        digitalWrite(pin, HIGH);
        delay(500);
        digitalWrite(pin, LOW);
        delay(500);
    }
}

// Функция с возвращаемым значением
float readVoltage(int pin) {
    int sensorValue = analogRead(pin);
    float voltage = sensorValue * (5.0 / 1023.0);
    return voltage;
}

void setup() {
    pinMode(13, OUTPUT);
    Serial.begin(9600);
}

void loop() {
    blinkLED();  // Вызов простой функции
    blinkTimes(13, 3);  // Вызов функции с параметрами
    
    float voltage = readVoltage(A0);  // Получение значения из функции
    Serial.println(voltage);
    
    delay(1000);
}
